<h1 align="center">Colorem</h1>

<p align="center"> Personalize your terminal with so many colors! </p>

<h1 align="center">How to use?</h1>

```js
const color = require("colorem");

console.log(color.red("red"))
console.log(color.black("black"))
console.log(color.cyan("cyan"))
console.log(color.yellow("yellow"))
console.log(color.green("green"))
console.log(color.magenta("magenta"))
console.log(color.white("white"))
console.log(color.blue("blue"))

console.log(color.BgRed("Bgred"))
console.log(color.BgBlack("Bgblack"))
console.log(color.BgCyan("Bgcyan"))
console.log(color.BgYellow("Bgyellow"))
console.log(color.BgGreen("Bggreen"))
console.log(color.BgMagenta("Bgmagenta"))
console.log(color.BgWhite("Bgwhite"))
console.log(color.BgBlue("Bgblue"))

console.log(color.bright("bright"))
console.log(color.underscore("underscore"))
console.log(color.dim("dim"))
console.log(color.blink("blink"))
console.log(color.reverse("reverse"))
console.log(color.hidden("hidden"))
```
